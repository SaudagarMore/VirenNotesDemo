abstract class Course {
    String courseName;
    int duration;
    Course(String courseName, int duration) {
        this.courseName = courseName;
        this.duration = duration;
    }
    abstract double calculateFee();
}

interface Certifiable {
    void issueCertificate();
}

class Overloading extends Course implements Certifiable {
    String platform;
    double baseFee;

    Overloading(String courseName, int duration, String platform, double baseFee) {
        super(courseName, duration);
        this.platform = platform;
        this.baseFee = baseFee;
    }
public static void main(String[] args) {
        Overloading overloading=new Overloading("Core Java", 3, "online",3000);
        System.out.println(overloading.calculateFee());
        overloading.issueCertificate();
        overloading.display();
        }
    double calculateFee() {
        return baseFee + baseFee * 0.10;
    }
    public void issueCertificate() {
        if (duration > 4) {
            System.out.println("Certificate Issued for " + courseName);
        }
        else{
            System.out.println("Not Eligible for Course Certificate");
        }
    }
    void display() {
        System.out.println("Course: " + courseName + ", Duration: " + duration + " weeks, Platform: " + platform + ", Final Fee: " + calculateFee());
    }
}
