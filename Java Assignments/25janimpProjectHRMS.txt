üîπ Generate Unique Random SignIn ID
@Service
public class AttendanceService {
    @Autowired
    private AttendanceRepository attendanceRepository;
    private String generateUniqueSignInId() {
        String signInId;
        do {
            signInId = "B" + UUID.randomUUID().toString().substring(0, 7).toUpperCase();
        } while (attendanceRepository.existsBySignInId(signInId));

        return signInId;
    }
üîπ SIGN IN Logic
    public void signIn(Attendance attendance) {
        attendance.setSignInId(generateUniqueSignInId());
        attendanceRepository.save(attendance);
    }
üîπ SIGN OUT Logic
    public void signOut(Attendance signOutRequest) {

        Attendance attendance = attendanceRepository
                .findByEmployeeIdAndCurrentDate(
                        signOutRequest.getEmployeeId(),
                        signOutRequest.getCurrentDate())
                .orElseThrow(() -> new RuntimeException("SignIn not found"));

        attendance.setSignOutTime(signOutRequest.getSignOutTime());
        attendance.setSignOutLocation(signOutRequest.getSignOutLocation());
        attendance.setIsSignOut(signOutRequest.isIsSignOut());
        attendance.setIsLateOut(signOutRequest.getIsLateOut());
        attendance.setIsFirstOut(signOutRequest.getIsFirstOut());

        attendanceRepository.save(attendance);
    }
}

@RestController
@RequestMapping("/api/attendance")
public class AttendanceController {
    @Autowired
    private AttendanceService attendanceService;
    // SIGN IN
    @PostMapping("/signIn")
    public ResponseEntity<String> signIn(@RequestBody Attendance attendance) {
        try {
            attendanceService.signIn(attendance);
            return ResponseEntity.status(HttpStatus.CREATED)
                    .body("SignIn successfully!");
        } catch (Exception e) {
            return ResponseEntity.status(HttpStatus.INTERNAL_SERVER_ERROR)
                    .body("Failed to add SignIn data.");
        }
    }
    // SIGN OUT
    @PostMapping("/signOut")
    public ResponseEntity<String> signOut(@RequestBody Attendance attendance) {
        try {
            attendanceService.signOut(attendance);
            return ResponseEntity.status(HttpStatus.CREATED)
                    .body("SignOut successfully!");
        } catch (Exception e) {
            return ResponseEntity.status(HttpStatus.INTERNAL_SERVER_ERROR)
                    .body("Failed to add SignOut data.");
        }
   }
}
